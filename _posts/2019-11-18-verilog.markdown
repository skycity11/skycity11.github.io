# 1 verilog常用小模块
## 1.1 上升沿及下降沿检测
```verilog
reg		pulze_r1;
reg		pulze_r2;
wire	pulze_rise;
wire	pulze_fall;

assign pulze_rise = ~pulze_r2 & pulze_r1;	//检测上升沿
assign pulze_fall = pulze_r2 & ~pulze_r1;	//检测下降沿

always @(posedge clk_50m or negedge rst_n) begin
	if(!rst_n) begin
		pulze_r1 <= 1'b0;
		pulze_r2 <= 1'b0;
	end
	else begin
		pulze_r1 <= pulze;
		pulze_r2 <= pulze_r1;
	end
end
```
